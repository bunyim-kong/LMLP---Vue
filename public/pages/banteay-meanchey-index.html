សូមស្វាគមន៍មកកាន់
ខេត្តបន្ទាយមានជ័យ

ព័ត៍មានអំពីខេត្តបន្ទាយមានជ័យ

ខេត្តបន្ទាយមានជ័យ មានទីតាំងស្ថិតនៅភាគពាយ័ព្យនៃប្រទេសកម្ពុជា និងមានទីរួមខេត្តឈ្មោះក្រុងសិរីសោភ័ណ។ ខេត្តនេះមានព្រំប្រទល់ជាប់នឹងប្រទេសថៃនៅភាគខាងលិច ដែលធ្វើឱ្យក្រុងប៉ោយប៉ែត ក្លាយជាច្រកពាណិជ្ជកម្មដ៏សំខាន់ និងមានសក្តានុពលសេដ្ឋកិច្ចខ្លាំង។ ឈ្មោះ "បន្ទាយមានជ័យ" មានន័យថា "បន្ទាយនៃជ័យជម្នះ"។ ខេត្តនេះត្រូវបានបង្កើតឡើងនៅឆ្នាំ ១៩៨៨ ដោយកាត់យកស្រុកមួយចំនួនចេញពីខេត្តបាត់ដំបង និងខេត្តសៀមរាប។ ខេត្តបន្ទាយមានជ័យមិនត្រឹមតែជាតំបន់កសិកម្មសំខាន់ប៉ុណ្ណោះទេ ថែមទាំងមានតំបន់ទេសចរណ៍ប្រវត្តិសាស្ត្រ និងធម្មជាតិជាច្រើនផងដែរ រួមមានប្រាសាទបុរាណដូចជា ប្រាសាទបន្ទាយឆ្មារ ប្រាសាទបន្ទាយទ័ព និងរមណីយដ្ឋានធម្មជាតិដូចជា អាងទឹកត្រពាំងថ្ម ដែលជាជម្រកសត្វស្លាបកម្រជាច្រើនប្រភេទផងដែរ។

តំបន់ប្រាសាទបុរាណ

ប្រាសាទបន្ទាយឆ្មារ

ប្រាសាទបន្ទាយឆ្មារ គឺមានរូបចម្លាក់ចម្រុះយ៉ាងសន្ធឹកសន្ធាប់ ដែលពណ៌នាតាំងពីទេវរូបព្រហ្មញ្ញសាសនា មានរូបព្រះព្រហ្ម ព្រះនារាយណ៍ ឬព្រះវិស្ណុ និងតួអង្គសំខាន់ទៀតនៃសាសនានោះ ។

ប្រាសាទបន្ទាយទ័ព

ស្ថិតនៅស្រុកថ្មពួក កសាងដោយព្រះបាទជ័យវរ្ម័នទី៧ នៅចុងសតវត្សរ៍ទី១២។ ប្រាសាទនេះទាក់ទងនឹងការលើកទ័ពរបស់ព្រះអង្គ។

ប្រាសាទប្រាំ

ស្ថិតនៅស្រុកភ្នំស្រុក កសាងនៅសតវត្សរ៍ទី១១ ឧទ្ទិសដល់សាសនាហិណ្ឌូ។ ប្រាសាទនេះមានប្រាង្គចំនួន៥ ទើបហៅថា "ប្រាសាទប្រាំ"។

ប្រាសាទមេបុណ្យ

ស្ថិតនៅក្នុងខេត្តបន្ទាយមានជ័យ កសាងនៅពាក់កណ្ដាលសតវត្សរ៍ទី១០ ដោយព្រះបាទរាជេន្ទ្រវរ្ម័នទី១ សម្រាប់ឧទ្ទិសដល់ព្រះឥសូរនិងដូនតា។

តំបន់ធម្មជាតិ និងភ្នំ

អាងទឹកត្រពាំងថ្ម

អាងត្រពាំងថ្ម (ATT) គឺជាកន្លែងមើលបក្សីកំពូលមួយនៅជិតខេត្តសៀមរាប។ វាជាដីចំណីសម្រាប់សត្វក្រៀល Sarus (Grus Antigone) នៅរដូវប្រាំង។ សត្វស្លាបប្រហែល 200 ប្រភេទត្រូវបានគេប្រទះឃើញនៅកន្លែងបំរុង ដែលទាំងនេះរួមមាន សត្វទីទុយឈើ សត្វទីទុយ បូព៌ា សត្វឥន្ទ្រីធំ សត្វទាសិត ខ្លែងខ្មៅ មីលគី ស្តេកលាបពណ៌ និង ស្តូកខ្មៅ។

ល្អាងភ្នំទូច

ល្អាងភ្នំទូច ជារមណីយដ្ឋានធម្មជាតិ និងវប្បធម៌ ដែលមានទីតាំងនៅស្រុក សិរីសោភ័ណ តាមបណ្តោយផ្លូវជាតិលេខ៥ ចម្ងាយប្រហែល ២៨ គីឡូម៉ែត្រពីទីរួមខេត្ត។ ទីតាំងនេះមានអណ្តូងធម្មជាតិ និងភ្នំធំៗ និងតូចដែលមានមុខ និងផ្ទៃថ្មធំៗ។ ពីលើកំពូលភ្នំ ភ្ញៀវទេសចរអាចគយគន់ទេសភាពព្រៃធម្មជាតិ និងវត្តមួយនៅសងខាងភ្នំ។
