សូមស្វាគមន៍មកកាន់
ខេត្តស្ទឹងត្រែង

ព័ត៍មានអំពីខេត្តស្ទឹងត្រែង

ខេត្តស្ទឹងត្រែង មានព្រំប្រទល់ជាប់ខេត្តរតនគិរី ខាងកើតជាប់ខេត្តមណ្ឌលគិរី និងក្រចេះ ខាងត្បូងជាប់ខេត្តកំពង់ធំ និងព្រះវិហារ នៅខាងលិច។ ព្រំប្រទល់ខាងជើងរបស់វាគឺព្រំដែនអន្តរជាតិរបស់កម្ពុជាជាមួយឡាវ។ ទន្លេមេគង្គហូរកាត់ខេត្ត។ ខេត្តនេះភាគច្រើនគ្របដណ្តប់ដោយព្រៃឈើ ប៉ុន្តែការកាប់ឈើ និងការនេសាទបានធ្វើឱ្យមានសម្ពាធខ្ពស់លើព្រៃឈើ និងធនធានជលផល។

កន្លែងកម្សាន្ត​

ប្រាសាទ​ភ្នំធាតុ

ប្រាសាទ​ភ្នំធាតុ គឺ​ជា​ប្រាសាទ​មួយ​មាន​វ័យ​ចំណាស់​រាប់​រយ​ឆ្នាំ ដែល​អ្នក​ខេត្ត​ស្ទឹងត្រែង​តែង​មាន​ជំនឿ​ថា ពូកែ​ស័ក្ដិសិទ្ធិ មាន​បារមី​ថែរក្សា ចាំ​ការពារ សង់​ឡើង​ដើម្បី​ដាក់​ធាតុ​សព​របស់​ស្ដេច បាន​ពលិកម្ម​ដើម្បី​ជាតិ​មាតុភូមិ។។ តាមទំនៀមទំលាប់ ពួកគេជឿថា ប្រាសាទនេះត្រូវបានសង់ឡើង ដើម្បីដាក់ធាតុសពរបស់ស្តេចមួយអង្គ ដែលបានពលិកម្មអស់ពីជីវិត ដើម្បីការពារជាតិ និងមាតុភូមិ។

ប្រាសាទព្រះគោ

ប្រាសាទព្រះគោបែរមុខទៅទិសខាងកើត។ ប្រាសាទនេះមានទំហំ៥០០ម៉ែត្រគុណ៤០០ម៉ែត្រមានកំពែងធ្វើអំពីថ្មបាយក្រៀមព័ទ្ធជុំវិញជាពីរដាច់ដោយឡែកពីគ្នា និងមានកំពែងមួយ។ ស្ថាបត្យកម្មរបស់ប្រាសាទបង្ហាញពីជំនាញ និងភាពច្នៃប្រឌិតរបស់ជនបុរាណ ដោយមានការតុបតែងលម្អលើថ្ម បង្ហាញពីរូបចម្លាក់នៃទេវតា និងរឿងព្រេងព្រហ្មញ្ញសាសនា។

តំបន់កម្សាន្ត

កោះខ្សាច់

កោះខ្សាច់ មានទីតាំងនៅលើដងទន្លេសេកុង។ តំបន់នេះជាប្រភេទរមណីយដ្ឋានធម្មជាតិតាមរដូវ គឺអាចទៅលេងកម្សាន្តបាននៅរដូវប្រាំង។ តំបន់នេះមានឆ្នេរខ្សាច់ល្អស្អាតមានប្រវែង ភ្ញៀវទេសចរអាចទៅងូតទឹកកម្សាន្តលេងតាមឆ្នេរខ្សាច់។

ទឹកធ្លាក់ព្រះនិមិ្មត
